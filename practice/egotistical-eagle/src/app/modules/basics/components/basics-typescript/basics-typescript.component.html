TypeScript

TypeScript: superset of JS; more features (Types, Classes, Interfaces, ...); strong typing (TS) vs dynamic typing (JS); must be compiled to JavaScript 

//--- docs
// http://www.typescriptlang.org/Handbook

//--- strong typing
let myString: string;
myString = 'This is a string';
myString = 4; // => error!

//--- TypeScript can also infer types
let anotherString = 'This is a string without :string';
anotherString = 4; // => error!

// can only infer values when values assigned at declaration
let yetAnotherString;
yetAnotherString = 'This is a string'; // TypeScript does not know type
yetAnotherString = 5' // => no error

//--- other basic types
let aString: string;
let aNumber: number;
let aBoolean: boolean;
let anArray: Array<string> // generic type => may only hold strings in this case
let anything: any; // use this rarely!
// we also got void (=> nothing) and enums (a set of numeric values)


//--- classes
// classes allow us to create 'blueprints' for objects
// in Angular, we use classes to create components, directives, pipes, ...

class Car {
  engineName: string;
  gears: number;
  private speed: number;

  constructor(speed: number) {
    this.speed = speed || 0;
  }

  accelerate(): void {
    this.speed++;
  }

  throttle(): void {
    this.speed--;
  }

  getSpeed(): void {
    console.log(this.speed);
  }

  static numberOfWheels(): number {
    return 4;
  }
} 

// instantiate (create) an object from a class
let car = new Car(5);
car.accelerate();
car.getSpeed();

// calling static method
console.log(Car.numberOfWheels()); 


//--- interfaces
// interfaces allow us to create contracts other classes/objects have to implement
// can use them to define custom types without creating classes
// are NOT compiled to JS - it's just for checking/validation done by TS compiler

interface User {
  username: string;
  password: string;
  confirmPassword?: string; // ? => optional property
}

let user: User;
user = { anything: 'anything', anynumber: 5 } // => error!
user = { username: 'max', password: 'secret' }

// interfaces can also contain functions w/o body, as it is only a blueprint/requirement

interface CanDrive {
  accelerate(speed: number): void;
}

let car: CanDrive = {
  accelerate: function(speed: number) {
    // ...
  }
}


//--- generics
// generics are types which can hold/use several types

let numberArray: Array<string> // this array will only accept numbers
numberArray = ['test']; // => error!
numberArray = [1, 2, 3];
